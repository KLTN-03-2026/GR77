model KycSession {
  id           String    @id @default(uuid())
  userId       String    @map("user_id")
  provider     String    @db.VarChar(50) // "sumsub" | "mock"
  status       KycStatus @default(NOT_STARTED)

  // provider ids
  externalRef  String?   @map("external_ref") @db.VarChar(100)
  reviewResult String?   @map("review_result") @db.VarChar(50)
  rejectReason String?   @map("reject_reason") @db.Text

  // extracted data
  extractedFullName String?   @map("extracted_full_name")
  extractedDob      DateTime? @map("extracted_dob")
  extractedDocNo    String?   @map("extracted_doc_no")

  createdAt    DateTime  @default(now()) @map("created_at")
  updatedAt    DateTime  @updatedAt @map("updated_at")

  user      User           @relation(fields: [userId], references: [id], onDelete: Cascade)
  documents KycDocument[]

  @@index([userId, status])
  @@index([externalRef])
  @@map("kyc_sessions")
}

model KycDocument {
  id           String   @id @default(uuid())
  kycSessionId String   @map("kyc_session_id")
  type         String   @db.VarChar(30) // "ID_FRONT" | "ID_BACK" | "SELFIE" | "VIDEO"
  fileUrl      String   @map("file_url") @db.Text
  createdAt    DateTime @default(now()) @map("created_at")

  session KycSession @relation(fields: [kycSessionId], references: [id], onDelete: Cascade)

  @@index([kycSessionId, type])
  @@map("kyc_documents")
}
